139:   /**
140:    * Checks if a request to the API is authenticated
141:    * Gets API Token from HTTP Request and return Auth result
142:    * 
143:    * @param  SS_HTTPRequest   $request    HTTP API request
144:    * @return array                        authentication result:

151:   public function authenticate(SS_HTTPRequest $request)
152:   {
153:     //get the token
154:     $token = $request->getHeader("X-Silverstripe-Apitoken");
155:     if (!$token)
156:     {
157:       $token = $request->requestVar('token');
158:     }
159: 
160:     if ( $token )

162:       //check token validity
163:       return $this->validateAPIToken( $token );
164:     }
165:     else{
166:       //no token, bad news
167:       return array(
168:         'valid'   => false,
169:         'message' => 'Token invalid.',
170:         'code'    => self::AUTH_CODE_TOKEN_INVALID

176:   /**
177:    * Validate the API token
178:    * 
179:    * @param  SS_HTTPRequest   $request    HTTP request with API token header "X-Silverstripe-Apitoken" or 'token' request var
180:    * @return array                        Result and eventual error message (valid, message, code)
181:    */
182:   private function validateAPIToken(string $token)
183:   {
184:     //get Member with that token
185:     $member = Member::get()->filter(array('ApiToken' => $token))->first();
186:     if ( $member )
187:     {
188:       //check token expiry
189:       $tokenExpire  = $member->ApiTokenExpire;
190:       $now          = time();
191:       $life         = Config::inst()->get( 'RESTfulAPI_TokenAuthenticator', 'tokenLife', Config::INHERITED );
192: 
193:       if ( $tokenExpire > ($now - $life) )
194:       {
195:         //all good, log Member in
196:         $member->logIn();
197: 
198:         return array(
199:           'valid'   => true,
200:           'message' => 'Token valid.',
201:           'code'    => self::AUTH_CODE_LOGGED_IN
202:         );
203:       }
204:       else{
205:         //too old
206:         return array(

208:           'message' => 'Token expired.',
209:           'code'    => self::AUTH_CODE_TOKEN_EXPIRED

212:     }
