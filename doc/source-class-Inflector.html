241: /**
242:  * Cache inflected values, and return if already available
243:  *
244:  * @param string $type Inflection type
245:  * @param string $key Original value
246:  * @param string $value Inflected value
247:  * @return string Inflected value, from cache
248:  */
249:     protected static function _cache($type, $key, $value = false) {
250:         $key = '_' . $key;
251:         $type = '_' . $type;
252:         if ($value !== false) {
253:             self::$_cache[$type][$key] = $value;
254:             return $value;
255:         }
256:         if (!isset(self::$_cache[$type][$key])) {
257:             return false;
258:         }
259:         return self::$_cache[$type][$key];
260:     }

262: /**
263:  * Clears Inflectors inflected value caches. And resets the inflection
264:  * rules to the initial values.
265:  *
266:  * @return void
267:  */
268:     public static function reset() {
269:         if (empty(self::$_initialState)) {
270:             self::$_initialState = get_class_vars('Inflector');
271:             return;
272:         }
273:         foreach (self::$_initialState as $key => $val) {
274:             if ($key !== '_initialState') {
275:                 self::${$key} = $val;
276:             }
277:         }

280: /**
281:  * Adds custom inflection $rules, of either 'plural', 'singular' or 'transliteration' $type.
282:  *
283:  * ### Usage:
284:  *
285:  * {{{
286:  * Inflector::rules('plural', array('/^(inflect)or$/i' => '\1ables'));
287:  * Inflector::rules('plural', array(
288:  *     'rules' => array('/^(inflect)ors$/i' => '\1ables'),
289:  *     'uninflected' => array('dontinflectme'),
290:  *     'irregular' => array('red' => 'redlings')
291:  * ));
292:  * Inflector::rules('transliteration', array('/å/' => 'aa'));
293:  * }}}
294:  *
295:  * @param string $type The type of inflection, either 'plural', 'singular' or 'transliteration'
296:  * @param array $rules Array of rules to be added.
297:  * @param boolean $reset If true, will unset default inflections for all
298:  *        new rules that are being defined in $rules.
299:  * @return void
300:  */
301:     public static function rules($type, $rules, $reset = false) {
302:         $var = '_' . $type;
303: 
304:         switch ($type) {
305:             case 'transliteration':
306:                 if ($reset) {
307:                     self::$_transliteration = $rules;
308:                 } else {
309:                     self::$_transliteration = $rules + self::$_transliteration;
310:                 }
311:             break;
312: 
313:             default:
314:                 foreach ($rules as $rule => $pattern) {
315:                     if (is_array($pattern)) {
316:                         if ($reset) {
317:                             self::${$var}[$rule] = $pattern;
318:                         } else {
319:                             if ($rule === 'uninflected') {
320:                                 self::${$var}[$rule] = array_merge($pattern, self::${$var}[$rule]);
321:                             } else {
322:                                 self::${$var}[$rule] = $pattern + self::${$var}[$rule];
323:                             }
324:                         }
325:                         unset($rules[$rule], self::${$var}['cache' . ucfirst($rule)]);
326:                         if (isset(self::${$var}['merged'][$rule])) {
327:                             unset(self::${$var}['merged'][$rule]);
328:                         }
329:                         if ($type === 'plural') {
330:                             self::$_cache['pluralize'] = self::$_cache['tableize'] = array();
331:                         } elseif ($type === 'singular') {
332:                             self::$_cache['singularize'] = array();
333:                         }
334:                     }
335:                 }
336:                 self::${$var}['rules'] = $rules + self::${$var}['rules'];
337:             break;
338:         }
339:     }

341: /**
342:  * Return $word in plural form.
343:  *
344:  * @param string $word Word in singular
345:  * @return string Word in plural
346:  * @link http://book.cakephp.org/2.0/en/core-utility-libraries/inflector.html#Inflector::pluralize
347:  */
348:     public static function pluralize($word) {
349:         if (isset(self::$_cache['pluralize'][$word])) {
350:             return self::$_cache['pluralize'][$word];
351:         }
352: 
353:         if (!isset(self::$_plural['merged']['irregular'])) {
354:             self::$_plural['merged']['irregular'] = self::$_plural['irregular'];
355:         }
356: 
357:         if (!isset(self::$_plural['merged']['uninflected'])) {
358:             self::$_plural['merged']['uninflected'] = array_merge(self::$_plural['uninflected'], self::$_uninflected);
359:         }
360: 
361:         if (!isset(self::$_plural['cacheUninflected']) || !isset(self::$_plural['cacheIrregular'])) {
362:             self::$_plural['cacheUninflected'] = '(?:' . implode('|', self::$_plural['merged']['uninflected']) . ')';
363:             self::$_plural['cacheIrregular'] = '(?:' . implode('|', array_keys(self::$_plural['merged']['irregular'])) . ')';
364:         }
365: 
366:         if (preg_match('/(.*)\\b(' . self::$_plural['cacheIrregular'] . ')$/i', $word, $regs)) {
367:             self::$_cache['pluralize'][$word] = $regs[1] . substr($word, 0, 1) . substr(self::$_plural['merged']['irregular'][strtolower($regs[2])], 1);
368:             return self::$_cache['pluralize'][$word];
369:         }
370: 
371:         if (preg_match('/^(' . self::$_plural['cacheUninflected'] . ')$/i', $word, $regs)) {
372:             self::$_cache['pluralize'][$word] = $word;
373:             return $word;
374:         }
375: 
376:         foreach (self::$_plural['rules'] as $rule => $replacement) {
377:             if (preg_match($rule, $word)) {
378:                 self::$_cache['pluralize'][$word] = preg_replace($rule, $replacement, $word);
379:                 return self::$_cache['pluralize'][$word];
380:             }
381:         }
382:     }

384: /**
385:  * Return $word in singular form.
386:  *
387:  * @param string $word Word in plural
388:  * @return string Word in singular
389:  * @link http://book.cakephp.org/2.0/en/core-utility-libraries/inflector.html#Inflector::singularize
390:  */
391:     public static function singularize($word) {
392:         if (isset(self::$_cache['singularize'][$word])) {
393:             return self::$_cache['singularize'][$word];
394:         }
395: 
396:         if (!isset(self::$_singular['merged']['uninflected'])) {
397:             self::$_singular['merged']['uninflected'] = array_merge(
398:                 self::$_singular['uninflected'],
399:                 self::$_uninflected
400:             );
401:         }
402: 
403:         if (!isset(self::$_singular['merged']['irregular'])) {
404:             self::$_singular['merged']['irregular'] = array_merge(
405:                 self::$_singular['irregular'],
406:                 array_flip(self::$_plural['irregular'])
407:             );
408:         }
409: 
410:         if (!isset(self::$_singular['cacheUninflected']) || !isset(self::$_singular['cacheIrregular'])) {
411:             self::$_singular['cacheUninflected'] = '(?:' . implode('|', self::$_singular['merged']['uninflected']) . ')';
412:             self::$_singular['cacheIrregular'] = '(?:' . implode('|', array_keys(self::$_singular['merged']['irregular'])) . ')';
413:         }
414: 
415:         if (preg_match('/(.*)\\b(' . self::$_singular['cacheIrregular'] . ')$/i', $word, $regs)) {
416:             self::$_cache['singularize'][$word] = $regs[1] . substr($word, 0, 1) . substr(self::$_singular['merged']['irregular'][strtolower($regs[2])], 1);
417:             return self::$_cache['singularize'][$word];
418:         }
419: 
420:         if (preg_match('/^(' . self::$_singular['cacheUninflected'] . ')$/i', $word, $regs)) {
421:             self::$_cache['singularize'][$word] = $word;
422:             return $word;
423:         }
424: 
425:         foreach (self::$_singular['rules'] as $rule => $replacement) {
426:             if (preg_match($rule, $word)) {
427:                 self::$_cache['singularize'][$word] = preg_replace($rule, $replacement, $word);
428:                 return self::$_cache['singularize'][$word];
429:             }
430:         }
431:         self::$_cache['singularize'][$word] = $word;
432:         return $word;
433:     }
